```typescript
import { NextRequest, NextResponse } from "next/server";

export async function POST(req: NextRequest) {
  const { actions, decision } = await req.json();

  const esUrl = process.env.ELASTICSEARCH_URL;
  const apiKey = process.env.API_KEY;

  if (!esUrl || !apiKey) {
    return NextResponse.json(
      { error: "Missing ELASTICSEARCH_URL or API_KEY" },
      { status: 500 }
    );
  }

  if (decision === "rejected") {
    // Index a rejection record
    await fetch(`${esUrl}/ic-notifications/_doc`, {
      method: "POST",
      headers: {
        Authorization: `ApiKey ${apiKey}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        "@timestamp": new Date().toISOString(),
        type: "remediation_override",
        message: "Operator overrode all proposed remediation actions",
        action: "override",
        severity: "info",
      }),
    });
    return NextResponse.json({ status: "rejected", executed: [] });
  }

  // Execute each approved action by indexing to ic-notifications
  const results = [];
  for (const action of actions) {
    const doc = buildRemediationDoc(action);
    const resp = await fetch(`${esUrl}/ic-notifications/_doc?refresh=wait_for`, {
      method: "POST",
      headers: {
        Authorization: `ApiKey ${apiKey}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify(doc),
    });
    results.push({
      actionId: action.id,
      toolId: action.toolId,
      success: resp.ok,
      docId: resp.ok ? (await resp.json())._id : null,
    });
  }

  return NextResponse.json({ status: "approved", executed: results });
}

function buildRemediationDoc(action: { toolId: string; params?: Record<string, string> }) {
  const now = new Date().toISOString();
  switch (action.toolId) {
    case "ic-notify-slack":
    case "notify-slack":
      return {
        "@timestamp": now,
        type: "slack_notification",
        channel: action.params?.channel || "#incidents",
        severity: action.params?.severity || "P2",
        message: action.params?.message || "Incident notification sent via Incident Cortex",
      };
    case "ic-create-jira-ticket":
    case "create-jira-ticket":
      return {
        "@timestamp": now,
        type: "jira_ticket",
        "ticket.id": `IC-${Date.now()}`,
        "ticket.url": `https://jira.example.com/browse/IC-${Date.now()}`,
        message: action.params?.title || "Incident ticket created by Incident Cortex",
        severity: action.params?.severity || "P2",
      };
    case "ic-rollback-deployment":
    case "rollback-deployment":
      return {
        "@timestamp": now,
        type: "deployment_rollback",
        action: "rollback",
        "deployment.service": action.params?.service || "payment-service",
        "deployment.version": action.params?.to_version || "v2.3.9",
        message: `Rollback initiated: ${action.params?.service || "payment-service"} to ${action.params?.to_version || "v2.3.9"}`,
      };
    case "ic-block-ip":
    case "block-ip":
      return {
        "@timestamp": now,
        type: "ip_block",
        action: "block",
        "target.ip": action.params?.ip_address || "203.0.113.42",
        message: `IP blocked: ${action.params?.ip_address || "203.0.113.42"} - ${action.params?.reason || "Suspicious activity"}`,
      };
    case "ic-index-incident-report":
    case "index-incident-report":
      return {
        "@timestamp": now,
        type: "incident_report",
        message: action.params?.summary || "Incident report generated by Incident Cortex",
        severity: action.params?.severity || "P2",
      };
    default:
      return {
        "@timestamp": now,
        type: "remediation_action",
        action: action.toolId,
        message: `Remediation action executed: ${action.toolId}`,
      };
  }
}
```

"use client";

import { useState } from "react";
import type { RemediationAction, RemediationResult } from "@/lib/types";
import { TOOL_LABELS } from "@/lib/constants";

interface RemediationCardProps {
  actions: RemediationAction[];
}

export function RemediationCard({ actions }: RemediationCardProps) {
  const [decision, setDecision] = useState<"pending" | "executing" | "approved" | "rejected">(
    "pending"
  );
  const [selected, setSelected] = useState<Set<string>>(
    () => new Set(actions.map((a) => a.id))
  );
  const [results, setResults] = useState<RemediationResult[]>([]);

  const toggle = (id: string) => {
    setSelected((prev) => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  };

  const execute = async (approve: boolean) => {
    if (!approve) {
      setDecision("executing");
      try {
        await fetch("/api/remediate", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ actions: [], decision: "rejected" }),
        });
      } catch {
        // Override still recorded locally even if API fails
      }
      setDecision("rejected");
      return;
    }

    setDecision("executing");
    const approvedActions = actions.filter((a) => selected.has(a.id));
    try {
      const resp = await fetch("/api/remediate", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ actions: approvedActions, decision: "approved" }),
      });
      const data = await resp.json();
      setResults(data.executed || []);
      setDecision("approved");
    } catch {
      setDecision("approved"); // Show approved even if API fails (optimistic)
    }
  };

  const borderColor =
    decision === "approved"
      ? "border-[#00BFB3]"
      : decision === "rejected"
      ? "border-[#FF6666]"
      : decision === "executing"
      ? "border-[#45B7D1]"
      : "border-[#FEC514]";

  return (
    <div
      className={`mt-3 rounded-lg border ${borderColor} bg-[#1D1E24] p-4 space-y-3`}
    >
      {decision === "pending" && (
        <>
          <h4 className="text-sm font-semibold text-[#FEC514]">
            Proposed Remediation Actions
          </h4>
          <div className="space-y-2">
            {actions.map((action) => (
              <label
                key={action.id}
                className="flex items-center gap-2 text-sm text-[#DFE5EF] cursor-pointer"
              >
                <input
                  type="checkbox"
                  checked={selected.has(action.id)}
                  onChange={() => toggle(action.id)}
                  className="accent-[#00BFB3] w-4 h-4"
                />
                <span>{TOOL_LABELS[action.toolId] ?? action.label}</span>
              </label>
            ))}
          </div>
          <div className="flex gap-2 pt-1">
            <button
              onClick={() => execute(true)}
              disabled={selected.size === 0}
              className="px-3 py-1.5 rounded text-xs font-medium bg-[#00BFB3] text-black hover:bg-[#00D9CA] transition-colors disabled:opacity-50"
            >
              Approve {selected.size === actions.length ? "All" : `(${selected.size})`}
            </button>
            <button
              onClick={() => execute(false)}
              className="px-3 py-1.5 rounded text-xs font-medium bg-[#343741] text-[#98A2B3] hover:bg-[#3d3e47] transition-colors"
            >
              Override
            </button>
          </div>
        </>
      )}
      {decision === "executing" && (
        <div className="flex items-center gap-2">
          <div className="w-4 h-4 border-2 border-[#45B7D1] border-t-transparent rounded-full animate-spin" />
          <p className="text-sm text-[#45B7D1] font-medium">
            Executing remediation actions...
          </p>
        </div>
      )}
      {decision === "approved" && (
        <div className="space-y-2">
          <p className="text-sm text-[#00BFB3] font-medium">
            Remediation approved &mdash; {results.length} action{results.length !== 1 ? "s" : ""} executed
          </p>
          {results.length > 0 && (
            <div className="space-y-1">
              {results.map((r) => (
                <div key={r.actionId} className="flex items-center gap-2 text-xs text-[#98A2B3]">
                  <span className={r.success ? "text-[#00BFB3]" : "text-[#FF6666]"}>
                    {r.success ? "\u2713" : "\u2717"}
                  </span>
                  <span>{TOOL_LABELS[r.toolId] ?? r.toolId}</span>
                  {r.docId && (
                    <span className="text-[#69707D] font-mono text-[10px]">{r.docId.slice(0, 8)}</span>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      )}
      {decision === "rejected" && (
        <p className="text-sm text-[#FF6666] font-medium">
          Remediation overridden &mdash; no actions taken
        </p>
      )}
    </div>
  );
}

### Task 6: Upgrade ic-detect-metric-anomaly with INLINE STATS

**Files:**
- Modify: `tools/tool_definitions.json` (ic-detect-metric-anomaly query)

**Step 1: Update the ES|QL query**

In `tools/tool_definitions.json`, change `ic-detect-metric-anomaly` query from:

```
FROM ic-metrics | WHERE host.name == ?hostName AND @timestamp >= ?startTime AND @timestamp <= ?endTime | STATS avg_cpu = AVG(system.cpu.total.pct), max_cpu = MAX(system.cpu.total.pct), avg_mem = AVG(system.memory.used.pct), avg_latency = AVG(http.response.latency_ms) BY ts_bucket = BUCKET(@timestamp, 5 minutes) | SORT ts_bucket
```

To (using INLINE STATS for single-query outlier detection):

```
FROM ic-metrics | WHERE host.name == ?hostName AND @timestamp >= ?startTime AND @timestamp <= ?endTime | INLINE STATS overall_avg_cpu = AVG(system.cpu.total.pct), overall_avg_latency = AVG(http.response.latency_ms) | STATS avg_cpu = AVG(system.cpu.total.pct), max_cpu = MAX(system.cpu.total.pct), avg_latency = AVG(http.response.latency_ms), overall_avg_cpu = VALUES(overall_avg_cpu), overall_avg_latency = VALUES(overall_avg_latency) BY ts_bucket = BUCKET(@timestamp, 5 minutes) | EVAL cpu_anomaly = max_cpu > overall_avg_cpu * 1.5, latency_anomaly = avg_latency > overall_avg_latency * 2 | SORT ts_bucket
```

This adds global averages alongside per-bucket stats, then flags anomalous buckets in one query (no two-step process).

Also update the description to mention INLINE STATS:
